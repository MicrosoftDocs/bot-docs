122  public override IForm<JObject> Build(Assembly resourceAssembly = null, string resourceName = null)
123  {
124  if (!_form.Fields.Any())
125  {
126  // No fieldss means add default field and confirmation
127  AddRemainingFields();
128  Confirm(new PromptAttribute(Configuration.Template(TemplateUsage.Confirmation)));
129  }
130  // Build all code into a single assembly and cache because assemblies have no GC.
131  var builder = new StringBuilder("switch (choice) {");
132  int choice = 1;
133  var entries = new List<CallScript>();
134  lock (_scripts)
135  {
136  foreach (var entry in _scripts)
137  {
138  if (entry.Value.Script == null)
139  {
140  entry.Value.Choice = choice++;
141  builder.Append($"\ncase {entry.Value.Choice}: {{{entry.Key}}}; break;");
142  entries.Add(entry.Value);
143  }
144  }
145  if (entries.Any())
146  {
147  // Define does not need to return a result.
148  builder.Append("\n}\nreturn null;");
149  var fun = Compile<ScriptGlobals, object>(builder.ToString());
150  foreach (var entry in entries)
151  {
152  entry.Script = fun;
153  }
154  }
155  }
156  return base.Build(resourceAssembly, resourceName);
157  }

243  private void ProcessOptions()
244  {
245  JToken references;
246  var assemblies = new List<string>() { "Microsoft.Bot.Builder.dll" };
247  if (_schema.TryGetValue("References", out references))
248  {
249  foreach (JToken reference in references.Children())
250  {
251  assemblies.Add((string)reference);
252  }
253  }
254  JToken importsChildren;
255  var imports = new List<string>();
256  if (_schema.TryGetValue("Imports", out importsChildren))
257  {
258  foreach (JToken import in importsChildren.Children())
259  {
260  imports.Add((string)import);
261  }
262  }
263  var dir = Path.GetDirectoryName(new Uri(Assembly.GetExecutingAssembly().CodeBase).LocalPath);
264  _options = CodeAnalysis.Scripting.ScriptOptions.Default
265  .AddReferences((from assembly in assemblies select System.IO.Path.Combine(dir, assembly)).ToArray())
266  .AddImports("Microsoft.Bot.Builder", "Microsoft.Bot.Builder.Dialogs",
267  "Microsoft.Bot.Builder.FormFlow", "Microsoft.Bot.Builder.FormFlow.Advanced",
268  "System.Collections.Generic", "System.Linq")
269  .AddImports(imports.ToArray());
270  }

320  private Exception CompileException(CompilationErrorException ex, string code)
321  {
322  Exception result = ex;
323  var match = System.Text.RegularExpressions.Regex.Match(ex.Message, @"\(\s*(?<line>\d+)\s*,\s*(?<column>\d+)\s*\)\s*:\s*(?<message>.*)");
324  if (match.Success)
325  {
326  var lineNumber = int.Parse(match.Groups["line"].Value) - 1;
327  var column = int.Parse(match.Groups["column"].Value) - 1;
328  var line = code.Split('\n')[lineNumber];
329  var minCol = Math.Max(0, column - 20);
330  var maxCol = Math.Min(line.Length, column + 20);
331  var msg = line.Substring(minCol, column - minCol) + "^" + line.Substring(column, maxCol - column);
332  result = new ArgumentException(match.Groups["message"].Value + ": " + msg);
333  }
334  return result;
335  }

337  private void AddSteps(IEnumerable<FieldJson.MessageOrConfirmation> steps)
338  {
339  foreach (var step in steps)
340  {
341  var active = ActiveScript(null, step.ActiveScript);
342  if (step.IsMessage)
343  {
344  if (step.MessageScript != null)
345  {
346  Message(MessageScript(step.MessageScript), active, step.Dependencies);
347  }
348  else
349  {
350  Message(step.Prompt, active, step.Dependencies);
351  }
352  }
353  else
354  {

356  {
357  Confirm(MessageScript(step.MessageScript), active, step.Dependencies);
358  }
359  else

361  Confirm(step.Prompt, active, step.Dependencies);
362  }

367  private void Fields(JObject schema, string prefix, IList<string> fields)
368  {
369  if (schema["properties"] != null)
370  {
371  foreach (JProperty property in schema["properties"])
372  {
373  var path = (prefix == null ? property.Name : $"{prefix}.{property.Name}");
374  var childSchema = (JObject)property.Value;
375  var eltSchema = FieldJson.ElementSchema(childSchema);
376  if (FieldJson.IsPrimitiveType(eltSchema))
377  {
378  fields.Add(path);
379  }
380  else
381  {
382  Fields(childSchema, path, fields);
383  }
384  }

388  private void FieldPaths(Type type, string path, List<string> paths)
389  {
390  var newPath = (path == "" ? path : path + ".");
391  foreach (var field in type.GetFields(BindingFlags.Public | BindingFlags.Instance))
392  {
393  TypePaths(field.FieldType, newPath + field.Name, paths);
394  }
395 
396  foreach (var property in type.GetProperties(BindingFlags.Public | BindingFlags.Instance))
397  {
398  if (property.CanRead && property.CanWrite)
399  {
400  TypePaths(property.PropertyType, newPath + property.Name, paths);
401  }
402  }
403  }

405  private void TypePaths(Type type, string path, List<string> paths)
406  {
407  if (type.IsClass)
408  {
409  if (type == typeof(string))
410  {
411  paths.Add(path);
412  }
413  else if (type.IsIEnumerable())
414  {
415  var elt = type.GetGenericElementType();
416  if (elt.IsEnum)
417  {
418  paths.Add(path);
419  }
420  else
421  {
422  // TODO: What to do about enumerations of things other than enums?
423  }
424  }
425  else
426  {
427  FieldPaths(type, path, paths);
428  }
429  }
430  else if (type.IsEnum)
431  {
432  paths.Add(path);
433  }
434  else if (type == typeof(bool))
435  {
436  paths.Add(path);
437  }
438  else if (type.IsIntegral())
439  {
440  paths.Add(path);
441  }
442  else if (type.IsDouble())
443  {
444  paths.Add(path);
445  }
446  else if (type.IsNullable() && type.IsValueType)
447  {
448  paths.Add(path);
449  }
450  else if (type == typeof(DateTime))
451  {
452  paths.Add(path);
453  }
